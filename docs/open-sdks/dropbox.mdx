## Code Example

```typescript

    import {initSDK} from '@opensdks/runtime'
    import {dropboxSdkDef} from '@opensdks/sdk-dropbox'

    const dropbox = initSDK(dropboxSdkDef, {
      auth: {
        // for provider-specific auth
        bearer: `${process.env['DROPBOX_TOKEN']}`,
        // or for passthrough auth via OPENINT_API_KEY & end user resourceId
        openInt: {
          apiKey: `${process.env['OPENINT_API_KEY']}`,
          resourceId: END_USER_RESOURCE_ID,
        }
      }, params: { ... }
    })

    // set_profile_photo
    const res = await dropbox.POST('/2/account/set_profile_photo')
```
