## Code Example

```typescript

    import {initSDK} from '@opensdks/runtime'
    import {twilioSdkDef} from '@opensdks/sdk-twilio'

    const twilio = initSDK(twilioSdkDef, {
      auth: {
        // for provider-specific auth
        bearer: `${process.env['TWILIO_TOKEN']}`,
        // or for passthrough auth via OPENINT_API_KEY & end user resourceId
        openInt: {
          apiKey: `${process.env['OPENINT_API_KEY']}`,
          resourceId: END_USER_RESOURCE_ID,
        }
      }
    })

    
    const res = await twilio.SERVERS('/v1/Services/{ServiceSid}/AlphaSenders')
```
